Course 7: Introduction to the CSS Grid Challenges

display: grid;  // Turn any HTML element into a grid container
In CSS Grid, the parent element is referred to as the container and its children are called items.
.container {
  display: grid;
  grid-template-rows: 50px 50px;  // Adds two rows
  grid-template-columns: 50px 50px;  // Adds two columns
}
x: <Numbwe>fr; //sets the x to a fraction of the available space
grid-column-gap: 10px;
grid-gap: <num>..; grid-row-gap: <num>px; grid-column-gap: <num>px;  // Add gap
grid-column: x/y;  // X is begining and y is the end line
grid-row: x/y;
justify-self: center/start/end;  // Horizontal - Cell
align-self: center/start/end;    // Vertical   - Cell
justify-items: center/start/end; // Horizontal - Grid
align-items: center/start/end;   // Vertical - Grid
grid-template-areas:  // in grid container
  "header header header"
  "advert content content"
  "footer footer footer";
Every word in the code represents a cell and every pair of quotation marks represent a row. In addition to custom labels, you can use a period (.) to designate an empty cell in the grid.
grid-area: header;  // Shows how to use grid-template-areas in grid item
grid-area: horizontal line to start at / vertical line to start at / horizontal line to end at / vertical line to end at;
grid-template-columns: repeat(2, 1fr 50px) 20px;
Above statement translates to: grid-template-columns: 1fr 50px 1fr 50px 20px;
grid-template-columns: 100px minmax(50px, 200px);  // Define two column
repeat(auto-fill, minmax(60px, 1fr));  // auto-fill
repeat(auto-fit, minmax(60px, 1fr));  // auto-fit

function initAjax() {
  let ajax = false;
  if(window.XMLHttpRequest) {
    ajax = new XMLHttpRequest();
  } else if(window.ActiveXObject("Microsoft.XMLHTTP")) {
    ajax = new ActiveXObject("Microsoft.XMLHTTP");
  }
  return ajax;
}

function getDataByPost(elementTarget, stringDataSource) {
  let XMLHttpRequestObject = initAjax();
  let element = document.getElementById(elementTarget);
  element.textContent = "Ajax: trying to retrive message...";
  if(XMLHttpRequestObject) {
    XMLHttpRequestObject.overrideMimeType("text");
    XMLHttpRequestObject.open("POST", stringDataSource);
    XMLHttpRequestObject.setRequestHeader(
      'Content-Type', 'application/x-www-form-urlencoded');
    XMLHttpRequestObject.onreadystatechange = function() {
      if(XMLHttpRequestObject.readyStat == 4 &&
         XMLHttpRequestObject.status == 200) {
        element.textContent = XMLHttpRequestObject.responseText;
      }
    };
    XMLHttpRequestObject.send('data=' + stringDataSource);
  } else {
    let element = document.getElementById(elementTarget);
    element.textContent = "Warning: Ajax didn't work on your browser!";
  }
}

- Counters:
counter-name:  This is the name of the counter that needs to be created or incremented or printed. It can be any custom name as the developer wishes.
integer:  This integer is an optional value that when provided next to the counter name will represent the initial value of the counter (in counter-set , counter-reset properties) or the value by which the counter should be incremented (in counter-increment ).
none:  This is the initial value for all 3 counter-* properties. When this value is used for counter-increment , the value of none of the counters are affected. When this is used for the other two, no counter is created.

connector-string:  This represents the string that must be placed between the values of two different counter levels (like the "." in "2.1.1").

function deepClone(obj) {
	function clone(obj, traversedObjects) {
		var copy;
		// primitive types
		if(obj === null || typeof obj !== "object") {
			return obj;
		}
		// detect cycles
		for(var i = 0; i < traversedObjects.length; i++) {
			if(traversedObjects[i] === obj) {
				throw new Error("Cannot clone circular object.");
			}
		}
		// dates
		if(obj instanceof Date) {
			copy = new Date();
			copy.setTime(obj.getTime());
			return copy;
		}
		// arrays
		if(obj instanceof Array) {
			copy = [];
			for(var i = 0; i < obj.length; i++) {
				copy.push(clone(obj[i], traversedObjects.concat(obj)));
			}
			return copy;
		}
		// simple objects
		if(obj instanceof Object) {
			copy = {};
			for(var key in obj) {
				if(obj.hasOwnProperty(key)) {
					copy[key] = clone(obj[key], traversedObjects.concat(obj));
				}
			}
			return copy;
		}
		throw new Error("Not a cloneable object.");
	}
	return clone(obj, []);
}

- Iterable Object:
var myIterableObject = {};
// An Iterable object must define a method located at the Symbol.iterator key:
myIterableObject[Symbol.iterator] = function () {
	// The iterator should return an Iterator object
	return {
		// The Iterator object must implement a method, next()
		next: function () {
			// next must itself return an IteratorResult object
			if (!this.iterated) {
				this.iterated = true;
				// The IteratorResult object has two properties
				return {
					// whether the iteration is complete, and
					done: false,
					// the value of the current iteration
					value: 'One'
				};
			}
			return {
			// When iteration is complete, just the done property is needed
				done: true
			};
		},
		iterated: false
	};
};
